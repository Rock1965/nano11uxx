60 /*
61 When PIN as an input, the PIN_MODE control the input mode of all ports.
62 This includes the use of the on-chip pull-up/pull-down resistor feature
63 and a special open drain operating mode.
64 */
65 typedef enum PIN_INPUT_MODE {
66  // Pin has an on-chip pull-up resistor enabled
67  INTERNAL_PULL_UP = 2,
68  // enables the pull-up resistor if the pin is at a logic high
69  // enables the pull-down resistor if the pin is at a logic low
70  REPEATER = 3,
71  // Pin has neither pull-up nor pull-down resistor enabled
72  NEITHER = 0,
73  // Pin has an on-chip pull-down resistor enabled
74  INTERNAL_PULL_DOWN = 1
75 }PIN_INPUT_MODE_T;
76 
77 /*
78 When pin as an output, the open drain mode causes the pin to be pulled low normally
79 if it is configured as an output and the data value is 0.
80 If the data value is 1, the output drive of the pin is turned off
81  */
82 typedef enum PIN_OUTPUT_MODE {
83  // Pin is in the normal (not open drain) mode
84  NOT_OPEN = 0,
85  // Pin is in the open drain mode.
86  OPEN_DRAIN = 1
87 }PIN_OUTPUT_MODE_T;
