static const uint8_t led_scripts[] = { // bit0=LED1, bit1=LED2, bit2=LED3, bit4=LED4

};
//
// main task
//
int main(void) {
#ifdef DEBUG

#else

#endif

dbg.start();
#endif

// LED Demo (can be removed)
//

//
// your setup code here
//
CTimer t(TIMER0);
t.second(0.5); // set interval 0.5sec (500ms)
t.enable();
while(1) {
//
// LED Demo (can be removed)
//
t.wait();
port = led_scripts[i];
i = (i+1) < sizeof(led_scripts) ? i+1 : 0;
}
return 0 ;
}
